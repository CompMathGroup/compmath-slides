def aren(t, u):
    x, y, vx, vy = u
    m = 0.012277471; M = 1 - m
    Dm = ((x+m)**2+y**2)**(1.5);
    De = ((x-M)**2+y**2)**(1.5)
    return np.array([vx, vy,
                x+2*vy-M*(x+m)/Dm-m*(x-M)/De,
                y-2*vx-M*  y  /Dm-m*  y  /De])

def euler(f, tau, t, u):
    k1 = f(t, u)
    return u + tau * k1
euler.order = 1; euler.name = '  '

def midpoint(f, tau, t, u):
    k1 = f(t, u)
    k2 = f(t + tau/2, u + tau/2*k1)
    return u + tau * k2
midpoint.order = 2; midpoint.name = '   '

def rk4(f, tau, t, u):
    k1 = f(t, u)
    k2 = f(t + tau/2, u + tau/2*k1)
    k3 = f(t + tau/2, u + tau/2*k2)
    k4 = f(t + tau  , u + tau  *k3)
    return u + tau * (k1 + 2*k2 + 2*k3 + k4) / 6

def fixed_stepsize(f, y0, tmax, method, tau=0.1):
    t = 0; u = y0
    T = [0]; Y = [y0]
    name = method.name
    while t < tmax:
        #      tmax -  tau
        if t + tau > tmax: tau = tmax - t
        u = method(f, tau, t, u)
        t += tau;
        T.append(t)
        Y.append(u)
    print('%s,  : %d'%(name, len(T)-1))
    return np.array(T), np.array(Y)

def adaptive_stepsize(f, y0, tmax, method, tol, tau=0.1):
    t = 0; u = y0
    T = [0]; Y = [y0]
    p = method.order; name = method.name
    failed = 0 #   
    while t < tmax:
        if t + tau > tmax: tau = tmax - t
        u1 = method(f, tau, t, u)                 #  
        u2 = method(f, tau/2, t, u)        
        u2 = method(f, tau/2, t+tau/2, u2)     #  
        err = np.linalg.norm(u1-u2)/(1-2**-p) #  
        fac = (tol/err)**(1 / (p+1))       #  tau
        taunew = tau * min(2, max(0.25, 0.8 * fac))
        if err < tol:            #  ,  
            t += tau; u = u1            
            T.append(t); Y.append(u)
        else: #   ,     tau
            failed += 1
        tau = taunew
    print('%s,  : %d, : %d'%(name, len(T)-1, failed))
    return np.array(T), np.array(Y)

def newton(F, dFdx, x0):
    x = x0
    for it in range(50):
        dx = np.linalg.solve(dFdx(x), F(x))
        x -= dx
        if np.linalg.norm(dx) < 1e-12:
            return x
    print('   !')
    return x

def imeuler(f, h, t, u):
    E = np.eye(len(u))
    k1 = newton(
        lambda k: k - f(t+h, u+h*k), #  k-f(t+h,u+h*k) = 0
        lambda k: E - h*f(t+h, u+h*k, jac=True), #   k
        f(t, u)) #  
    return u + h*k1
