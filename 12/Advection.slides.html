import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

def solve_advection(u0, g, method, M=200, courant=0.5, verbose=False):
    X = np.linspace(0, 1, M+1); h = X[1] - X[0]
    if verbose: print(':', method.name, ',   c/h =', courant)
    tau = courant * h   #   = 1
    u = u0(X)           #   
    t = 0; tmax = 0.6; steps = 0
    while t < tmax:
        if t + tau > tmax: tau = tmax - t + 1e-14
        unext = np.empty_like(u)
        unext[0] = g(t+tau) #      
        #       
        unext[1:] = method(u, tau, h)
        u = unext       #      
        t += tau; steps += 1
    if verbose: print('t =', t, ', ', steps, '')
    return X, u

def upwind(u, tau, h):
    sigma = tau/h
    return sigma*u[:-1] + (1-sigma)*u[1:]
upwind.name = ' '

def central(u, tau, h):
    sigma = tau/h
    return np.append(
        u[1:-1] - sigma/2 * (u[2:] - u[:-2]),
